## Endpoint

```
GET /api/Usuario/Lista
```

Retrieves all users registered in the Sistema Venta application, including their role information and status.

## Response

The endpoint returns a `Response<List<UsuarioDTO>>` wrapper containing the list of users.

<ResponseField name="status" type="boolean" required>
  Indicates if the request was successful
</ResponseField>

<ResponseField name="value" type="UsuarioDTO[]">
  Array of user objects
  
  <Expandable title="UsuarioDTO properties">
    <ResponseField name="idUsuario" type="integer">
      Unique identifier for the user
    </ResponseField>
    
    <ResponseField name="nombreCompleto" type="string">
      Full name of the user
    </ResponseField>
    
    <ResponseField name="correo" type="string">
      Email address of the user
    </ResponseField>
    
    <ResponseField name="idRol" type="integer">
      Role identifier associated with the user
    </ResponseField>
    
    <ResponseField name="rolDescripcion" type="string">
      Description of the user's role
    </ResponseField>
    
    <ResponseField name="clave" type="string">
      User password (hashed)
    </ResponseField>
    
    <ResponseField name="esActivo" type="integer">
      Active status (1 = active, 0 = inactive)
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="msg" type="string">
  Error message if status is false
</ResponseField>
