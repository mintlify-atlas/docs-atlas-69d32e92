title: 'Register Sale'
api: 'POST /api/Venta/Registrar'
description: 'Register a new sale with its detail items'
---

## Request Body

The request accepts a sale object containing the sale header information and an array of detail items.

<ParamField body="NumeroDocumento" type="string" required>
  Document number for the sale (invoice or receipt number)
</ParamField>

<ParamField body="TipoPago" type="string" required>
  Payment type (e.g., "Efectivo", "Tarjeta", "Transferencia")
</ParamField>

<ParamField body="TotalTexto" type="string" required>
  Total amount as text (formatted string)
</ParamField>

<ParamField body="DetalleVenta" type="array" required>
  Array of sale detail items
  
  <Expandable title="DetalleVenta properties">
    <ParamField body="IdProducto" type="integer" required>
      Product identifier
    </ParamField>
    
    <ParamField body="DescripcionProducto" type="string" required>
      Product description or name
    </ParamField>
    
    <ParamField body="Cantidad" type="integer" required>
      Quantity of product sold
    </ParamField>
    
    <ParamField body="PrecioTexto" type="string" required>
      Unit price as text (formatted string)
    </ParamField>
    
    <ParamField body="TotalTexto" type="string" required>
      Line total as text (formatted string)
    </ParamField>
  </Expandable>
</ParamField>

## Response

<ResponseField name="status" type="boolean">
  Indicates whether the operation was successful
</ResponseField>

<ResponseField name="value" type="object">
  The registered sale object
  
  <Expandable title="VentaDTO properties">
    <ResponseField name="IdVenta" type="integer">
      Generated sale identifier
    </ResponseField>
    
    <ResponseField name="NumeroDocumento" type="string">
      Document number for the sale
    </ResponseField>
    
    <ResponseField name="TipoPago" type="string">
      Payment type
    </ResponseField>
    
    <ResponseField name="TotalTexto" type="string">
      Total amount as text
    </ResponseField>
    
    <ResponseField name="FechaRegistro" type="string">
      Registration date and time
    </ResponseField>
    
    <ResponseField name="DetalleVenta" type="array">
      Array of sale detail items
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="msg" type="string">
  Error message if status is false
</ResponseField>
