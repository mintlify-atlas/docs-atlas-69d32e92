---
title: 'Sales History'
api: 'GET /api/Venta/Historial'
description: 'Retrieve sales history filtered by sale number or date range'
---

## Query Parameters

<ParamField query="buscarPor" type="string" required>
  Search criteria type. Possible values:
  - `"numero"` - Search by sale number
  - `"fecha"` - Search by date range
</ParamField>

<ParamField query="numeroVenta" type="string">
  Sale document number to search for. Required when `buscarPor` is `"numero"`.
  
  Example: `"0001-00000123"`
</ParamField>

<ParamField query="fechaInicio" type="string">
  Start date for date range search. Required when `buscarPor` is `"fecha"`.
  
  Format: `dd/MM/yyyy` (e.g., `"01/03/2026"`)
</ParamField>

<ParamField query="fechaFin" type="string">
  End date for date range search. Required when `buscarPor` is `"fecha"`.
  
  Format: `dd/MM/yyyy` (e.g., `"05/03/2026"`)
</ParamField>

## Response

<ResponseField name="status" type="boolean">
  Indicates whether the operation was successful
</ResponseField>

<ResponseField name="value" type="array">
  Array of sale objects matching the search criteria
  
  <Expandable title="VentaDTO properties">
    <ResponseField name="IdVenta" type="integer">
      Sale identifier
    </ResponseField>
    
    <ResponseField name="NumeroDocumento" type="string">
      Document number for the sale
    </ResponseField>
    
    <ResponseField name="TipoPago" type="string">
      Payment type
    </ResponseField>
    
    <ResponseField name="TotalTexto" type="string">
      Total amount as text
    </ResponseField>
    
    <ResponseField name="FechaRegistro" type="string">
      Registration date and time
    </ResponseField>
    
    <ResponseField name="DetalleVenta" type="array">
      Array of sale detail items
      
      <Expandable title="DetalleVentaDTO properties">
        <ResponseField name="IdProducto" type="integer">
          Product identifier
        </ResponseField>
        
        <ResponseField name="DescripcionProducto" type="string">
          Product description or name
        </ResponseField>
        
        <ResponseField name="Cantidad" type="integer">
          Quantity sold
        </ResponseField>
        
        <ResponseField name="PrecioTexto" type="string">
          Unit price as text
        </ResponseField>
        
        <ResponseField name="TotalTexto" type="string">
          Line total as text
        </ResponseField>
      </Expandable>
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="msg" type="string">
  Error message if status is false
</ResponseField>
